\noindent
\verb+\ZAdd,\ZSub,\ZMul,\ZDiv,\ZDivRes+ ... 加減乗除

\begin{verbatim}
\def\x{7}\def\y{4}
\ZAdd\x\y\z$\x+\y=\z$,
\ZSub\x\y\z$\x-\y=\z$,
\ZMul\x\y\z$\x\times \y=\z$,
\ZDiv\x\y\z$\x/\y=\z$,
\ZDivRes\x\y\q\r$\x=\y\times \q+\r$
\end{verbatim}

\noindent
